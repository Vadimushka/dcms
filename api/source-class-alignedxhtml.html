 13:         if (is_null($this->SPACER)) {
 14:             $this->SPACER = "  ";
 15:         }
 16:         if (is_null($this->OFFSET)) {
 17:             $this->OFFSET = 0;
 18:         }
 19:         if (is_null($this->SKIPTAGS)) {
 20:             $this->SKIPTAGS = array('a', 'span', 'img', 'sup', 'sub');
 21:         }
 22:         /*
 23:           Теги <textarea>, <pre> и <script> - особенные, и с ними
 24:           придется попотеть.
 25:           Нужно защитить содержимое этих тегов от вмешательства
 26:           при выравнивании: убрать на время, во-первых, переводы строки,
 27:           во-вторых, HTML-теги, которые могут встретиться внутри
 28:           строковых переменных в скриптах
 29:           (точнее, не сами теги, а открывающие и закрывающие скобки).
 30:          */
 31:         $xhtml = str_replace(array("\x01", "\x02", "\x03"), '', $xhtml);
 32:         $xhtml = preg_replace_callback(
 33:                 '/
 34:             (<(textarea|script|pre)(?:[^>"\']*|"[^"]*"|\'[^\']*\')*>)
 35:             (.*?)
 36:             (<\/\2>)
 37:             /six',
 38:                 // модификатор 's' не забываем: точка тут должна совпадать 
 39:                 // и с символом новой строки; модификатор 'x' позволяет добавлять в шаблон
 40:                 // необрабатываемые пробелы и переводы строки, чтобы он лучше читался
 41:                 create_function(
 42:                         '$matches', '$tagbody = $matches[3];
 43:                 $tagbody = str_replace("\n", "\x01", $tagbody);
 44:                 $tagbody = str_replace("<", "\x02", $tagbody);
 45:                 $tagbody = str_replace(">", "\x03", $tagbody);
 46:                 return $matches[1] . $tagbody . $matches[4];'
 47:                 ), $xhtml);
 48: 
 49:         // регулярное выражение для HTML-тега 
 50:         // (модификатор s не нужен, т.к. точки в выражении нет)
 51:         $tagpattern = '/<(\/?)(\w+)(?:[^>"\']*|"[^"]*"|\'[^\']*\')*>/';
 52: 
 53:         // убираем переводы строки внутри тегов (заменяем на пробелы)
 54:         $xhtml = preg_replace_callback(
 55:                 $tagpattern, create_function(
 56:                         '$matches', 'return str_replace("\n", " ", $matches[0]);'
 57:                 ), $xhtml);
