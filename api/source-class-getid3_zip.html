319:     static function ZIPparseGeneralPurposeFlags($flagbytes, $compressionmethod) {
320:         $ParsedFlags['encrypted'] = (bool) ($flagbytes & 0x0001);
321: 
322:         switch ($compressionmethod) {
323:             case 6:
324:                 $ParsedFlags['dictionary_size']    = (($flagbytes & 0x0002) ? 8192 : 4096);
325:                 $ParsedFlags['shannon_fano_trees'] = (($flagbytes & 0x0004) ? 3    : 2);
326:                 break;
327: 
328:             case 8:
329:             case 9:
330:                 switch (($flagbytes & 0x0006) >> 1) {
331:                     case 0:
332:                         $ParsedFlags['compression_speed'] = 'normal';
333:                         break;
334:                     case 1:
335:                         $ParsedFlags['compression_speed'] = 'maximum';
336:                         break;
337:                     case 2:
338:                         $ParsedFlags['compression_speed'] = 'fast';

395:     static function DOStime2UNIXtime($DOSdate, $DOStime) {
396:         // wFatDate
397:         // Specifies the MS-DOS date. The date is a packed 16-bit value with the following format:
398:         // Bits      Contents
399:         // 0-4    Day of the month (1-31)
400:         // 5-8    Month (1 = January, 2 = February, and so on)
401:         // 9-15   Year offset from 1980 (add 1980 to get actual year)
402: 
403:         $UNIXday    =  ($DOSdate & 0x001F);
404:         $UNIXmonth  = (($DOSdate & 0x01E0) >> 5);
405:         $UNIXyear   = (($DOSdate & 0xFE00) >> 9) + 1980;
406: 
407:         // wFatTime
408:         // Specifies the MS-DOS time. The time is a packed 16-bit value with the following format:
409:         // Bits   Contents
410:         // 0-4    Second divided by 2
411:         // 5-10   Minute (0-59)
412:         // 11-15  Hour (0-23 on a 24-hour clock)
413: 
414:         $UNIXsecond =  ($DOStime & 0x001F) * 2;
415:         $UNIXminute = (($DOStime & 0x07E0) >> 5);
416:         $UNIXhour   = (($DOStime & 0xF800) >> 11);
417: 
418:         return gmmktime($UNIXhour, $UNIXminute, $UNIXsecond, $UNIXmonth, $UNIXday, $UNIXyear);
