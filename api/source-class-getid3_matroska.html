1188:     private function readEBMLint()
1189:     {
1190:         $actual_offset = $this->current_offset - $this->EBMLbuffer_offset;
1191: 
1192:         // get length of integer
1193:         $first_byte_int = ord($this->EBMLbuffer[$actual_offset]);
1194:         if (0x80 & $first_byte_int) {
1195:             $length = 1;
1196:         } elseif (0x40 & $first_byte_int) {
1197:             $length = 2;
1198:         } elseif (0x20 & $first_byte_int) {
1199:             $length = 3;
1200:         } elseif (0x10 & $first_byte_int) {
1201:             $length = 4;
1202:         } elseif (0x08 & $first_byte_int) {
1203:             $length = 5;
1204:         } elseif (0x04 & $first_byte_int) {
1205:             $length = 6;
1206:         } elseif (0x02 & $first_byte_int) {
1207:             $length = 7;
1208:         } elseif (0x01 & $first_byte_int) {
1209:             $length = 8;
1210:         } else {
1211:             throw new Exception('invalid EBML integer (leading 0x00) at '.$this->current_offset);

1214:         // read
1215:         $int_value = self::EBML2Int(substr($this->EBMLbuffer, $actual_offset, $length));
1216:         $this->current_offset += $length;
1217: 
1218:         return $int_value;

1379:     private static function EBML2Int($EBMLstring) {
1380:         // http://matroska.org/specs/
1381: 
1382:         // Element ID coded with an UTF-8 like system:
1383:         // 1xxx xxxx                                  - Class A IDs (2^7 -2 possible values) (base 0x8X)
1384:         // 01xx xxxx  xxxx xxxx                       - Class B IDs (2^14-2 possible values) (base 0x4X 0xXX)
1385:         // 001x xxxx  xxxx xxxx  xxxx xxxx            - Class C IDs (2^21-2 possible values) (base 0x2X 0xXX 0xXX)
1386:         // 0001 xxxx  xxxx xxxx  xxxx xxxx  xxxx xxxx - Class D IDs (2^28-2 possible values) (base 0x1X 0xXX 0xXX 0xXX)
1387:         // Values with all x at 0 and 1 are reserved (hence the -2).
1388: 
1389:         // Data size, in octets, is also coded with an UTF-8 like system :
1390:         // 1xxx xxxx                                                                              - value 0 to  2^7-2
1391:         // 01xx xxxx  xxxx xxxx                                                                   - value 0 to 2^14-2
1392:         // 001x xxxx  xxxx xxxx  xxxx xxxx                                                        - value 0 to 2^21-2
1393:         // 0001 xxxx  xxxx xxxx  xxxx xxxx  xxxx xxxx                                             - value 0 to 2^28-2
1394:         // 0000 1xxx  xxxx xxxx  xxxx xxxx  xxxx xxxx  xxxx xxxx                                  - value 0 to 2^35-2
1395:         // 0000 01xx  xxxx xxxx  xxxx xxxx  xxxx xxxx  xxxx xxxx  xxxx xxxx                       - value 0 to 2^42-2
1396:         // 0000 001x  xxxx xxxx  xxxx xxxx  xxxx xxxx  xxxx xxxx  xxxx xxxx  xxxx xxxx            - value 0 to 2^49-2
