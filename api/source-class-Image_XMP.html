 79:     /**
 80:     * Reads all the JPEG header segments from an JPEG image file into an array
 81:     *
 82:     * @param string $filename - the filename of the JPEG file to read
 83:     * @return array $headerdata - Array of JPEG header segments
 84:     * @return boolean FALSE - if headers could not be read
 85:     */
 86:     function _get_jpeg_header_data($filename)
 87:     {
 88:         // prevent refresh from aborting file operations and hosing file
 89:         ignore_user_abort(true);
 90: 
 91:         // Attempt to open the jpeg file - the at symbol supresses the error message about
 92:         // not being able to open files. The file_exists would have been used, but it
 93:         // does not work with files fetched over http or ftp.
 94:         if (is_readable($filename) && is_file($filename) && ($filehnd = fopen($filename, 'rb'))) {
 95:             // great
 96:         } else {
 97:             return false;
 98:         }
 99: 
100:         // Read the first two characters
101:         $data = fread($filehnd, 2);
102: 
103:         // Check that the first two characters are 0xFF 0xD8  (SOI - Start of image)
104:         if ($data != "\xFF\xD8")
105:         {
106:             // No SOI (FF D8) at start of file - This probably isn't a JPEG file - close file and return;
107:             echo '<p>This probably is not a JPEG file</p>'."\n";
108:             fclose($filehnd);
109:             return false;
110:         }
111: 
112:         // Read the third character
113:         $data = fread($filehnd, 2);
114: 
115:         // Check that the third character is 0xFF (Start of first segment header)
116:         if ($data{0} != "\xFF")
117:         {
118:             // NO FF found - close file and return - JPEG is probably corrupted
119:             fclose($filehnd);
120:             return false;
121:         }
122: 
123:         // Flag that we havent yet hit the compressed image data
124:         $hit_compressed_image_data = false;
125: 
126:         // Cycle through the file until, one of: 1) an EOI (End of image) marker is hit,
127:         //                                       2) we have hit the compressed image data (no more headers are allowed after data)
128:         //                                       3) or end of file is hit
129: 
130:         while (($data{1} != "\xD9") && (!$hit_compressed_image_data) && (!feof($filehnd)))
131:         {
132:             // Found a segment to look at.
133:             // Check that the segment marker is not a Restart marker - restart markers don't have size or data after them
134:             if ((ord($data{1}) < 0xD0) || (ord($data{1}) > 0xD7))
135:             {
136:                 // Segment isn't a Restart marker
137:                 // Read the next two bytes (size)
138:                 $sizestr = fread($filehnd, 2);
139: 
140:                 // convert the size bytes to an integer
141:                 $decodedsize = unpack('nsize', $sizestr);
142: 
143:                 // Save the start position of the data
144:                 $segdatastart = ftell($filehnd);
145: 
146:                 // Read the segment data with length indicated by the previously read size
147:                 $segdata = fread($filehnd, $decodedsize['size'] - 2);
148: 
149:                 // Store the segment information in the output array
150:                 $headerdata[] = array(
151:                     'SegType'      => ord($data{1}),
152:                     'SegName'      => $GLOBALS['JPEG_Segment_Names'][ord($data{1})],
153:                     'SegDataStart' => $segdatastart,
154:                     'SegData'      => $segdata,
155:                 );
156:             }
157: 
158:             // If this is a SOS (Start Of Scan) segment, then there is no more header data - the compressed image data follows
159:             if ($data{1} == "\xDA")
160:             {
161:                 // Flag that we have hit the compressed image data - exit loop as no more headers available.
162:                 $hit_compressed_image_data = true;
163:             }
164:             else
165:             {
166:                 // Not an SOS - Read the next two bytes - should be the segment marker for the next segment
167:                 $data = fread($filehnd, 2);
168: 
169:                 // Check that the first byte of the two is 0xFF as it should be for a marker
170:                 if ($data{0} != "\xFF")
171:                 {
172:                     // NO FF found - close file and return - JPEG is probably corrupted
173:                     fclose($filehnd);
174:                     return false;
175:                 }
176:             }
177:         }
178: 
179:         // Close File
180:         fclose($filehnd);
181:         // Alow the user to abort from now on
182:         ignore_user_abort(false);
183: 
184:         // Return the header data retrieved
185:         return $headerdata;
186:     }

189:     /**
190:     * Retrieves XMP information from an APP1 JPEG segment and returns the raw XML text as a string.
191:     *
192:     * @param string $filename - the filename of the JPEG file to read
193:     * @return string $xmp_data - the string of raw XML text
194:     * @return boolean FALSE - if an APP 1 XMP segment could not be found, or if an error occured
195:     */
196:     function _get_XMP_text($filename)
197:     {
198:         //Get JPEG header data
199:         $jpeg_header_data = $this->_get_jpeg_header_data($filename);
200: 
201:         //Cycle through the header segments
202:         for ($i = 0; $i < count($jpeg_header_data); $i++)
203:         {
204:             // If we find an APP1 header,
205:             if (strcmp($jpeg_header_data[$i]['SegName'], 'APP1') == 0)
206:             {
207:                 // And if it has the Adobe XMP/RDF label (http://ns.adobe.com/xap/1.0/\x00) ,
208:                 if (strncmp($jpeg_header_data[$i]['SegData'], 'http://ns.adobe.com/xap/1.0/'."\x00", 29) == 0)
209:                 {
210:                     // Found a XMP/RDF block
211:                     // Return the XMP text
212:                     $xmp_data = substr($jpeg_header_data[$i]['SegData'], 29);
213: 
214:                     return trim($xmp_data); // trim() should not be neccesary, but some files found in the wild with null-terminated block (known samples from Apple Aperture) causes problems elsewhere (see http://www.getid3.org/phpBB3/viewtopic.php?f=4&t=1153)
215:                 }
216:             }
217:         }
218:         return false;
219:     }

221:     /**
222:     * Parses a string containing XMP data (XML), and returns an array
223:     * which contains all the XMP (XML) information.
224:     *
225:     * @param string $xml_text - a string containing the XMP data (XML) to be parsed
226:     * @return array $xmp_array - an array containing all xmp details retrieved.
227:     * @return boolean FALSE - couldn't parse the XMP data
228:     */
229:     function read_XMP_array_from_text($xmltext)
230:     {
231:         // Check if there actually is any text to parse
232:         if (trim($xmltext) == '')
233:         {
234:             return false;
235:         }
236: 
237:         // Create an instance of a xml parser to parse the XML text
238:         $xml_parser = xml_parser_create('UTF-8');
239: 
240:         // Change: Fixed problem that caused the whitespace (especially newlines) to be destroyed when converting xml text to an xml array, as of revision 1.10
241: 
242:         // We would like to remove unneccessary white space, but this will also
243:         // remove things like newlines (&#xA;) in the XML values, so white space
244:         // will have to be removed later
245:         if (xml_parser_set_option($xml_parser, XML_OPTION_SKIP_WHITE, 0) == false)
246:         {
247:             // Error setting case folding - destroy the parser and return
248:             xml_parser_free($xml_parser);
249:             return false;
250:         }
251: 
252:         // to use XML code correctly we have to turn case folding
253:         // (uppercasing) off. XML is case sensitive and upper
254:         // casing is in reality XML standards violation
255:         if (xml_parser_set_option($xml_parser, XML_OPTION_CASE_FOLDING, 0) == false)
256:         {
257:             // Error setting case folding - destroy the parser and return
258:             xml_parser_free($xml_parser);
259:             return false;
260:         }
261: 
262:         // Parse the XML text into a array structure
263:         if (xml_parse_into_struct($xml_parser, $xmltext, $values, $tags) == 0)
264:         {
265:             // Error Parsing XML - destroy the parser and return
266:             xml_parser_free($xml_parser);
267:             return false;
268:         }
269: 
270:         // Destroy the xml parser
271:         xml_parser_free($xml_parser);
272: 
273:         // Clear the output array
274:         $xmp_array = array();
275: 
276:         // The XMP data has now been parsed into an array ...
277: 
278:         // Cycle through each of the array elements
279:         $current_property = ''; // current property being processed
280:         $container_index = -1; // -1 = no container open, otherwise index of container content
281:         foreach ($values as $xml_elem)
282:         {
283:             // Syntax and Class names
284:             switch ($xml_elem['tag'])
285:             {
286:                 case 'x:xmpmeta':
287:                     // only defined attribute is x:xmptk written by Adobe XMP Toolkit; value is the version of the toolkit
288:                     break;
289: 
290:                 case 'rdf:RDF':
291:                     // required element immediately within x:xmpmeta; no data here
292:                     break;
293: 
294:                 case 'rdf:Description':
295:                     switch ($xml_elem['type'])
296:                     {
297:                         case 'open':
298:                         case 'complete':
299:                             if (array_key_exists('attributes', $xml_elem))
300:                             {
301:                                 // rdf:Description may contain wanted attributes
302:                                 foreach (array_keys($xml_elem['attributes']) as $key)
303:                                 {
304:                                     // Check whether we want this details from this attribute
305:                                     if (in_array($key, $GLOBALS['XMP_tag_captions']))
306:                                     {
307:                                         // Attribute wanted
308:                                         $xmp_array[$key] = $xml_elem['attributes'][$key];
309:                                     }
310:                                 }
311:                             }
312:                         case 'cdata':
313:                         case 'close':
314:                             break;
315:                     }
316: 
317:                 case 'rdf:ID':
318:                 case 'rdf:nodeID':
319:                     // Attributes are ignored
320:                     break;
321: 
322:                 case 'rdf:li':
323:                     // Property member
324:                     if ($xml_elem['type'] == 'complete')
325:                     {
326:                         if (array_key_exists('attributes', $xml_elem))
327:                         {
328:                             // If Lang Alt (language alternatives) then ensure we take the default language
329:                             if (isset($xml_elem['attributes']['xml:lang']) && ($xml_elem['attributes']['xml:lang'] != 'x-default'))
330:                             {
331:                                 break;
332:                             }
333:                         }
334:                         if ($current_property != '')
335:                         {
336:                             $xmp_array[$current_property][$container_index] = (isset($xml_elem['value']) ? $xml_elem['value'] : '');
337:                             $container_index += 1;
338:                         }
339:                     //else unidentified attribute!!
340:                     }
341:                     break;
342: 
343:                 case 'rdf:Seq':
344:                 case 'rdf:Bag':
345:                 case 'rdf:Alt':
346:                     // Container found
347:                     switch ($xml_elem['type'])
348:                     {
349:                         case 'open':
350:                             $container_index = 0;
351:                             break;
352:                         case 'close':
353:                             $container_index = -1;
354:                             break;
355:                         case 'cdata':
356:                             break;
357:                     }
358:                     break;
359: 
360:                 default:
361:                     // Check whether we want the details from this attribute
362:                     if (in_array($xml_elem['tag'], $GLOBALS['XMP_tag_captions']))
363:                     {
364:                         switch ($xml_elem['type'])
365:                         {
366:                             case 'open':
367:                                 // open current element
368:                                 $current_property = $xml_elem['tag'];
369:                                 break;
370: 
371:                             case 'close':
372:                                 // close current element
373:                                 $current_property = '';
374:                                 break;
375: 
376:                             case 'complete':
377:                                 // store attribute value
378:                                 $xmp_array[$xml_elem['tag']] = (isset($xml_elem['value']) ? $xml_elem['value'] : '');
379:                                 break;
380: 
381:                             case 'cdata':
382:                                 // ignore
383:                                 break;
384:                         }
385:                     }
386:                     break;
387:             }
388: 
389:         }
390:         return $xmp_array;
391:     }
