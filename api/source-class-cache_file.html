129: abstract class cacher {
130: 
131:     protected static function _read($cache_name, $no_cache = false) {
132:         static $cache = array();
133:         if ($no_cache || !isset($cache[$cache_name])) {
134:             $cache[$cache_name] = cache::get($cache_name);
135:         }
136:         return $cache[$cache_name];
137:     }
138: 
139:     protected static function _max_ttl($cache) {
140:         //return 1000;
141:         $max_ttl = 0;
142:         foreach ($cache as $data) {
143:             $max_ttl = max($max_ttl, $data['t'] - TIME);
144:         }
145:         return $max_ttl;
146:     }
147: 
148:     public static function get($cache_name, $name) {
149: 
150:         if (!$cache = self::_read($cache_name)) {
151:             return false;
152:         }
153: 
154: 
155:         if (!isset($cache[$name])) {
156:             // нет такой переменной
157:             return false;
158:         }
159: 
160:         if ($cache[$name]['t'] < TIME) {
161:             // время жизни переменной вышло
162:             return false;
163:         }
164: 
165:         return $cache[$name]['v'];
166:     }
167: 
168:     /**
169:      * Удаление устаревших данных
170:      * @param type $cache
171:      */
172:     protected static function _clear(&$cache) {
173:         // удаление устаревших данных
174:         foreach ($cache as $name => $data) {
175:             if ($data['t'] >= TIME)
176:                 continue;
177:             unset($cache[$name]);
178:         }
179:     }
180: 
181:     public static function set($cache_name, $name, $val, $ttl = 0) {
182:         $cache = (array) self::_read($cache_name, true);
183:         self::_clear($cache);
184:         $cache[$name] = array('t' => $ttl + TIME, 'v' => $val);
185:         cache::set($cache_name, $cache, self::_max_ttl($cache));
186:         self::_read($cache_name, true);
187:         return true;
188:     }
