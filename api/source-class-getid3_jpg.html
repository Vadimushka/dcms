166:     function CastAsAppropriate($value) {
167:         if (is_array($value)) {
168:             return $value;

172:             return getid3_lib::CastAsInt($value);
173:         } elseif (preg_match('#^[0-9\.]+$#', $value)) {
174:             return (float) $value;
